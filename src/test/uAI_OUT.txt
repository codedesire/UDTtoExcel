TYPE "uAI_OUT"
VERSION : 0.1


  STRUCT
   Value : REAL ;	//Нормированное значение аналогового канала
   Val_Percent : REAL ;	//Значение аналогового канала в процентах
   Val_Mem : REAL ;	//Память аналогового значения. Для сглаживания
   Smooth_On : BOOL ;	//Включен режим сглаживания
   Break : BOOL ;	//"Обрыв"
   Over : BOOL ;	//"Короткое замыкание"
   NoGood : BOOL ;	//"Плохой сигнал" (вне 5% зоны, но еще не КЗ и не "обрыв")
   Display_On : BOOL ;	//Включен режим отображения границ на экране
   Bad : BOOL ;	//"Недостоверный сигнал" (вне 0-20мА, но еще в 5% зоне)
   Disabled : BOOL ;	//Отключен канал
   NoNet : BOOL ;	//Нет связи с модулем
   HH : BOOL ;	//Верхнее аварийное значение
   H : BOOL ;	//Верхнее предупредительное значение
   L : BOOL ;	//Нижнее предупредительное значение
   LL : BOOL ;	//Нижнее аварийное значение
   Vel_High : BOOL ;	//Сигнализация о высокой/низкой скорости изменения
   PLC_OB82 : BOOL ;	//Диагностическое событие
   PLC_OB83 : BOOL ;	//Модуль канала удален
   Subst_On : BOOL ;	//Включен режим замещения
  END_STRUCT ;
END_TYPE
